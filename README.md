# Find the first duplicate in a list

def first_duplicate(lst):
    seen = set()
    for num in lst:
        if num in seen:
            return num
        seen.add(num)
    return -1

# Test it out!
print(first_duplicate([1, 2, 3, 4, 2, 5, 6]))  # Expected output: 2
